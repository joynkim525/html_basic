Modify the class

case1) class_name.class_variable = 'new'

  • instance: changed
  • class: changed
case2) instance_name.class_variable = 'new'
  • instance: not changed
  • class: not changed
case3) class_name.instance_variable = 'new'
  • instance: changed
  • class: changed
case4) instance_name.instance_variable = 'new'
  • instance: changed
  • class: not dependet on instance
case5) class_name.instance_variable = 'new1' & instance_name.instance_variable = 'new2'
  • instance: changed to new2
  • class: changed to new1
case5) class_name.__class_variable = 'new' (__class_variable is fixed)
  • instance: not changed
  • class: changed
  • new instance: not changed
case6) instance_name.__class_variable = 'new' (__class_variable is fixed)
  • instance: not changed
  • class: not changed
  • new instance: not changed
